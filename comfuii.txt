dl() {
  local url="$1"
  local dest="$2"

  mkdir -p "$(dirname "$dest")"

  if [[ -f "$dest" ]]; then
    echo "âœ… Already exists: $(basename "$dest")"
    return 0
  fi

  echo "â¬‡ï¸  Downloading: $(basename "$dest")"

  if command -v curl >/dev/null 2>&1; then
    curl -L --fail --retry 5 --retry-delay 2 -o "$dest" "$url"
  else
    wget -c -O "$dest" "$url"
  fi
}
